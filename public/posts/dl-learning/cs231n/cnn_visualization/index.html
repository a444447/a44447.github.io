def jitter(X, ox, oy):
    """
    X: PyTorch Tensor of shape (N, C, H, W)
    ox, oy: Integers giving number of pixels to jitter along W and H axes
    """
    if ox != 0:
        left = X[:, :, :, :-ox]
        right = X[:, :, :, -ox:]
        X = torch.cat([right, left], dim=3)
    if oy != 0:
        top = X[:, :, :-oy]
        bottom = X[:, :, -oy:]
        X = torch.cat([bottom, top], dim=2)
    return X
